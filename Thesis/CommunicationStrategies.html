#Method for detecting reciprocal ties
def find_wanted_numbers(df, user, nmbrs):
    wanted_numbers = set()
    if len(df.loc[user]) > len(nmbrs) and len(nmbrs) > 0:
        for nmb in nmbrs:
            if ( (2 in df.loc[user,nmb]['type'].tolist() and 1 in df.loc[user,nmb]['type'].tolist()) and \
                  (df.loc[user,nmb]['duration'].sum() > 60) ):
                wanted_numbers.add(nmb)
    
    return wanted_numbers

deltas = {}
for user in wanted_users:
    pr = pd.DataFrame(pre_window[pre_window.user == user])
    pr.set_index(['user'], inplace=True)
    
    ob = pd.DataFrame(observation_window[observation_window.user == user])
    ob.set_index(['user'], inplace=True)
    
    po = pd.DataFrame(post_window[post_window.user == user])
    po.set_index(['user'], inplace=True)
    
    deltas[user] = []
    for tie in pre_window_ties[user]:
        dates = pr[pr.number==tie]['timestamp_datetime']
        for x in xrange(len(dates)-1):
            deltas[user].append((dates.tolist()[x+1] - dates.tolist()[x]).total_seconds())
    
    for tie in obs_window_ties[user]:
        dates_ob = ob[ob.number==tie]['timestamp_datetime']
        for x in xrange(len(dates_ob)-1):
            deltas[user].append((dates_ob.tolist()[x+1] - dates_ob.tolist()[x]).total_seconds())
            
    for tie in post_window_ties[user]:
        dates_po = po[po.number==tie]['timestamp_datetime']
        for x in xrange(len(dates_po)-1):
            deltas[user].append((dates_po.tolist()[x+1] - dates_po.tolist()[x]).total_seconds())
